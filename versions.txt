            entry.target.classList.add('show');
          }, i * 100);

      document.querySelectorAll(".parallax").forEach(el => {
        el.style.transform = `translate(${x}px,${y}px)`;
      });
    });

    /* CARD 3D EFFECT */
    document.querySelectorAll('.card').forEach(card => {
      card.addEventListener('mousemove', e => {
        const rect = card.getBoundingClientRect();
        const x = e.clientX - rect.left;
        const y = e.clientY - rect.top;

        const rotateX = (y / rect.height - 0.5) * 10;
        const rotateY = (x / rect.width - 0.5) * -10;

        card.style.transform = `rotateX(${rotateX}deg) rotateY(${rotateY}deg)`;
      });

      card.addEventListener('mouseleave', () => {
        card.style.transform = "rotateX(0) rotateY(0)";
      });

    window.onload = () => setTimeout(() => document.getElementById('loader').classList.add('hidden'), 900)

    const fadeObserver = new IntersectionObserver(entries => {
      entries.forEach(e => {
        if (e.isIntersecting) {
          e.target.classList.add('show');
          if (e.target.querySelector('.counter')) animateCounter(e.target.querySelector('.counter'))
        }
      })
    }, { threshold: .4 })
    document.querySelectorAll('.fade').forEach(el => fadeObserver.observe(el))

    const sectionObserver = new IntersectionObserver(entries => {
      entries.forEach(entry => {
        if (entry.isIntersecting) entry.target.classList.add('story-visible')
      })
    }, { threshold: .15 })
    document.querySelectorAll('section').forEach(sec => sectionObserver.observe(sec))

    document.querySelector('.toggle').onclick = () => document.body.classList.toggle('light')

    function animateCounter(el) {
      let target = +el.dataset.target, duration = 1400, startTime = null
      function ease(t) { return 1 - Math.pow(1 - t, 3) }
      function update(time) {
        if (!startTime) startTime = time
        let p = Math.min((time - startTime) / duration, 1)
        el.innerText = Math.floor(ease(p) * target)
        if (p < 1) requestAnimationFrame(update)
        else el.innerText = target
      }
      requestAnimationFrame(update)
    }

    function closeModal() { document.getElementById('modal').style.display = 'none' }
    function backdrop(e) { if (e.target.id === 'modal') closeModal() }
